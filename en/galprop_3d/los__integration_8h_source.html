213  //Check for empty rbins
214  if (fRbins.size() > 1) {
215  //Make sure they are monotonically increasing
216  if (fRbins[0] < 0) {
217  throw(LOSargumentError("Rbins must be >= 0"));
218  }
219  for ( size_t i = 0; i < fRbins.size()-1; ++i ) {
220  if ( fRbins[i] >= fRbins[i+1] )
221  throw(LOSargumentError("Rbins must be monotonically increasing"));

226  rmax = fRmax;

229  const double ymax = std::max(fabs(fymax), fabs(fymin));
230  rmax = sqrt(xmax*xmax+ymax*ymax);
231  }
232 
233  if (fRbins[fRbins.size()-1] < rmax) {
234  fnRbins = fRbins.size();
235  faddBoundaries = 1;
236  } else if (fRbins[0] > 0) {
237  fnRbins = fRbins.size();
238  faddBoundaries = 0;
239  } else {
240  fnRbins = fRbins.size() - 1;
241  faddBoundaries = 1;

244  fnRbins = 1;
245  fRbins.resize(1);

247  faddBoundaries = 0;
